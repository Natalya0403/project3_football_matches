#СЧИТАЕМ КУМУЛЯТИВНУЮ СТАТИСТИКУ ДЛЯ ДОМАШНИХ КОМАНД
# Сброс индексов (если данные после предыдущих операций имеют индексные уровни, которые могут влиять на порядок)
project_filtered = project_filtered.reset_index(drop=True)

# Сортировка по имени команды, сезону и дате
project_filtered.sort_values(by=['season', 'date'], ascending=[True, True], inplace=True)

#Кумулятивная статистика для домашних команд
#Кумулятивное количество матчей
project_filtered['home_total_matches'] = project_filtered.groupby(['home_team_name', 'season'])['home_team_result'].transform('cumcount') + 1

# Кумулятивные результаты домашних команд (победы, поражения, ничьи)
project_filtered['cumulative_home_wins'] = project_filtered.groupby(['home_team_name', 'season'])['home_team_result'].transform(lambda x: (x == 1).cumsum()) #победа
project_filtered['cumulative_home_losses'] = project_filtered.groupby(['home_team_name', 'season'])['home_team_result'].transform(lambda x: (x == -1).cumsum()) #поражение
project_filtered['cumulative_home_draws'] = project_filtered.groupby(['home_team_name', 'season'])['home_team_result'].transform(lambda x: (x == 0).cumsum()) #ничья
# Кумулятивные голы домашних команд
project_filtered['cumulative_home_goals_scored'] = project_filtered.groupby(['home_team_name', 'season'])['home_team_goal'].transform('cumsum') #забитые голы
project_filtered['cumulative_home_goals_conceded'] = project_filtered.groupby(['home_team_name', 'season'])['away_team_goal'].transform('cumsum') #пропущенные голы

# Считаем рейтинг домашней команды
project_filtered['home_rating'] = project_filtered['cumulative_home_wins'] * 3 + project_filtered['cumulative_home_draws'] *1 + (project_filtered['cumulative_home_goals_scored'] - project_filtered['cumulative_home_goals_conceded'])

# Функция для расчета статистики для последних 10 игр
def calculate_last_10_games_stats(team_name, historical_data):
    # Фильтруем данные для данной команды (как для домашней, так и для гостевой)
    home_games = historical_data[historical_data['home_team_name'] == team_name].tail(10)
    away_games = historical_data[historical_data['away_team_name'] == team_name].tail(10)

    # Рассчитываем статистики для последних 10 игр
    stats = {}

    # Форма домашних и гостевых команд
    stats['home_form'] = ((home_games['home_team_result'] == 1) * 3 + (home_games['home_team_result'] == 0) * 1).sum() / len(home_games) if not home_games.empty else 0
    stats['away_form'] = ((away_games['away_team_result'] == 1) * 3 + (away_games['away_team_result'] == 0) * 1).sum() / len(away_games) if not home_games.empty else 0

    # Характеристики команд
    stats['home_attack_rating'] = home_games['home_attack_rating'].mean() if not home_games.empty else 0
    stats['away_attack_rating'] = away_games['away_attack_rating'].mean() if not away_games.empty else 0
    stats['home_defense_rating'] = home_games['home_defense_rating'].mean() if not home_games.empty else 0
    stats['away_defense_rating'] = away_games['away_defense_rating'].mean() if not away_games.empty else 0
    stats['home_rating'] = home_games['home_rating'].mean() if not home_games.empty else 0
    stats['away_rating'] = away_games['away_rating'].mean() if not away_games.empty else 0

    # Кумулятивная статистика
    stats['cumulative_home_wins'] = (home_games['home_team_result'] == 1).sum() if not home_games.empty else 0
    stats['cumulative_away_losses'] = (away_games['away_team_result'] == -1).sum() if not away_games.empty else 0
    stats['cumulative_home_goals_scored'] = home_games['home_team_goal'].sum() if not home_games.empty else 0
    stats['cumulative_away_goals_conceded'] = away_games['home_team_goal'].sum() if not away_games.empty else 0
    stats['cumulative_home_losses'] = (home_games['home_team_result'] == -1).sum() if not home_games.empty else 0

    #Влияние домашнего поля на результат
    stats['home_vs_away_performance'] = (home_games['home_team_result'] == 1).sum() / len(home_games) if not home_games.empty else 0

    return stats

    total_home_matches=('home_team_name', 'count'),  # Количество матчей
    home_wins=('home_team_result', lambda x: (x == 1).sum()),  # Победы
    home_draws=('home_team_result', lambda x: (x == 0).sum()),  # Ничьи
    home_losses=('home_team_result', lambda x: (x == -1).sum()),  # Поражения
    home_goals_scored=('home_team_goal', 'sum'),  # Забитые голы
    home_goals_conceded=('away_team_goal', 'sum'),  # Пропущенные голы
).reset_index()
